useEffect(() => { async function handleResetLink() { try { // Get the URL that opened the app const url = await Linking.getInitialURL() if (!url) { setError('No reset link found. Please request a new one.') return } // Parse query params from the deep link const parsed = Linking.parse(url) const tokenHash = parsed.queryParams?.token_hash as string const type = parsed.queryParams?.type as string if (type === 'recovery' && tokenHash) { const { error } = await supabase.auth.verifyOtp({ token_hash: tokenHash, type: 'recovery', }) if (error) { setError('Reset link expired. Please request a new one.') } else { setReady(true) } } else { setError('Invalid reset link. Please request a new one.') } } catch (e) { setError('Something went wrong. Please request a new reset link.') } } handleResetLink() }, [])
